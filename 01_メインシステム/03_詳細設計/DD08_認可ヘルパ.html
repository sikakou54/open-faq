3. 詳細設計本文

3.1 認可ヘルパ実装

// app/workers/main-api/src/lib/authz.ts
export type ProjectRole = 'admin' | 'member';

export function requireTenant(c: Context, expectedTenantId: string) {
  const principal = c.get('principal');
  if (principal.contractOwnerUserId !== expectedTenantId) throw new HTTPException(404); // 404 偽装
}

export function requireOwner(c: Context) {
  const principal = c.get('principal');
  if (!principal.isOwner) throw new HTTPException(403); // E-AUTHZ-OWNER-ONLY
}

export async function requireProjectRole(c: Context, projectId: string, minRole: ProjectRole) {
  const principal = c.get('principal');
  const project = await c.env.DB.prepare(
    `SELECT contract_owner_user_id FROM projects WHERE id = ?1`
  ).bind(projectId).first<{ contract_owner_user_id: string }>();
  if (!project) throw new HTTPException(404);
  if (project.contract_owner_user_id !== principal.contractOwnerUserId) throw new HTTPException(404); // 404 偽装

  if (principal.isOwner) return; // オーナーは全プロジェクト bypass(暗黙の admin)

  const granted = principal.projectRoles.get(projectId);
  if (!granted) throw new HTTPException(404); // E-AUTHZ-PROJECT-DENIED(404 偽装)
  if (minRole === 'admin' && granted !== 'admin') throw new HTTPException(403); // E-AUTHZ-FORBIDDEN
  // minRole='member' は admin/member いずれでも通過
}

export async function requireInquiry(c: Context, inquiryId: string) {
  const principal = c.get('principal');
  const row = await c.env.DB.prepare(
    `SELECT contract_owner_user_id, project_id FROM inquiries WHERE id = ?1`
  ).bind(inquiryId).first<{ contract_owner_user_id: string; project_id: string }>();
  if (!row) throw new HTTPException(404);
  if (row.contract_owner_user_id !== principal.contractOwnerUserId) throw new HTTPException(404);
  await requireProjectRole(c, row.project_id, 'member');
}

export function requireRole(...roles: Role[]): MiddlewareHandler {
  return async (c, next) => {
    const principal = c.get('principal');
    if (!roles.includes(principal.role)) throw new HTTPException(403);
    await next();
  };
}

3.1a プロジェクトの質問数上限・アラート API の認可と上限解決

プロジェクト利用設定(SCR-021 / SCR-021-M1)は オーナー専有ではない。GET /projects/{id}/quota-limits は member、PATCH /projects/{id}/quota-limits/questions は admin で認可する。利用者向け API は質問数の上限とアラート閾値だけを参照・更新し、無料枠・FAQの設定ルートは提供しない。

質問数設定の 解決ヘルパは KV を一次参照し、ミス時は project_quota_limits.resource_kind='question_monthly_limit' を引く。優先順位で採用した行の threshold=null は明示的な上限OFFとして扱い、下位の運営者上書きやデフォルト上限へフォールバックしない。

export async function resolveProjectQuestionLimit(c: Context, projectId: string) {
  const cached = await c.env.KV.get(`usage-limit:${ownerOf(projectId)}:${projectId}`, 'json');
  if (cached?.question) return cached.question;
  const rows = await c.env.DB.prepare(
    `SELECT threshold, free_quota, alert_thresholds, source, valid_until FROM project_quota_limits
       WHERE project_id = ?1 AND resource_kind = 'question_monthly_limit' AND valid = 1`
  ).bind(projectId).all();
  const owner = rows.results.find(r => r.source === 'owner');
  const operator = rows.results.find(r => r.source === 'operator' /* && valid_until 未経過 */);
  return owner ?? operator ?? defaultQuestionLimitFromKV(c); // owner > operator > default
}

PATCH 成功時は質問数の source='owner' 行だけを upsert する。FAQ無料枠の内部解決は請求処理用ヘルパで「運営者上書き > デフォルト値」とし、SCR-021 の認可・レスポンス・更新処理から分離する。

3.1b メンバー割当解除の認可

export async function requireRemovableProjectGrant(c: Context, projectId: string, userId: string) {
  await requireProjectRole(c, projectId, 'admin');
  const principal = c.get('principal');
  if (principal.userId === userId) throw appError('E-AUTHZ-SELF-MUTATION');
  if (await isContractOwner(c.env.DB, userId)) throw appError('E-AUTHZ-OWNER-PROTECTED');

  await requireSameTenantUser(c, userId);
}

DELETE /projects/{id}/members/{userId} は requireRemovableProjectGrant を使用する。他プロジェクト有効割当数は認可エラーには使わず、割当解除後にアカウントを論理削除するかを決める業務処理で参照する。

3.2 Principal 構築（参照のみ）

c.get('principal') で取得する Principal の構築フローは ../02_基本設計/08_認証・認可設計.html §6.1 を正本とする。Principal には以下を含む。

  • userId: ユーザー ID(オーナー / プロジェクト管理者 / メンバー全員)
  • contractOwnerUserId: 所属契約の owner user_id(本人がオーナーなら自身の userId、メンバーなら所属契約の owner user_id)
  • actorRole: 操作者種別(管理者 = users 行 / 運営者 = 別 D1 / システム)。Principal 構築時に決定され、audit_logs.actor_role の値(admin / service_operator / system)に対応。ウィジェット利用者は管理コンソールPrincipalを構築しない
  • isOwner: オーナー判定(contract_owners 行存在で判定 = co.user_id IS NOT NULL)
  • projectRoles: Map<projectId, 'admin' | 'member'>。オーナー / メンバー共通で project_users WHERE valid=1 を 1 クエリで取得して構築する。認可ヘルパは isOwner=true のときに projectRoles の中身を参照せず bypass する優先順位を保つ。projectRoles は通知宛先解決と画面表示の事実情報として参照可能

Principal 構築クエリ(参考、基本設計と同期)

SELECT u.id, u.role, u.status, u.valid,
       (co.user_id IS NOT NULL)                          AS is_owner,
       COALESCE(co.user_id, pu.contract_owner_user_id)   AS contract_owner_user_id
FROM users u
LEFT JOIN contract_owners co ON co.user_id = u.id AND co.valid = 1
LEFT JOIN project_users   pu ON pu.user_id = u.id AND pu.valid = 1
WHERE u.id = ?1 AND u.valid = 1
LIMIT 1;

projectRoles は別途 SELECT project_id, role FROM project_users WHERE user_id=? AND valid=1 で取得する。

3.3 オーナー境界（NFR-301）

すべての CRUD API は contract_owner_user_id フィルタを必須 とする。requireTenant / requireProject / requireInquiry で境界チェックを実施し、他オーナーのデータには 404 を返す（漏洩防止のため 403 ではなく 404）。

3.4 不正アクセスの監査記録

境界違反検知時は authz.owner_boundary_violation / authz.project_boundary_violation を retention_class=general で記録する。詳細は DD09_監査ログ書込・完全性検証.html を参照。
