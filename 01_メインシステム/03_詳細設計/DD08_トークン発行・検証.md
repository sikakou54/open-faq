### 3.2 HMAC-SHA256 保存方式

```ts
// app/shared/src/lib/token.ts
export async function generateToken(
  purpose: TokenPurpose, payload: Record<string, string>, env: Env, ttlSec: number,
): Promise<string> {
  const tokenId = generateUlid();
  const random = crypto.getRandomValues(new Uint8Array(32));
  const rawToken = `${tokenId}.${base64urlEncode(random)}`;
  const tokenHash = await hmacSha256(env.MASTER_KEY, rawToken);

  await env.DB.prepare(`
    INSERT INTO access_tokens (id, token_hash, purpose, meta, created_at, expires_at, user_id)
    VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7)
  `).bind(tokenId, tokenHash, purpose, JSON.stringify(payload),
        new Date().toISOString(),
        new Date(Date.now() + ttlSec * 1000).toISOString(),
        payload.userId ?? null).run();

  return rawToken;  // ユーザーに返す生トークン (DB には hash のみ)
}

export async function verifyToken(
  rawToken: string, purpose: TokenPurpose, env: Env,
): Promise<{ payload: Record<string, string> }> {
  const tokenHash = await hmacSha256(env.MASTER_KEY, rawToken);
  const record = await env.DB.prepare(
    `SELECT id, purpose, meta, expires_at, used_at FROM access_tokens WHERE token_hash = ?1`
  ).bind(tokenHash).first<TokenRow>();
  if (!record) throw new HTTPException(400, { message: 'TOKEN_INVALID' });
  if (record.purpose !== purpose) throw new HTTPException(400, { message: 'TOKEN_INVALID' });
  if (record.used_at) throw new HTTPException(400, { message: 'TOKEN_REUSED' });
  if (new Date(record.expires_at).getTime() < Date.now()) {
    throw new HTTPException(400, { message: 'TOKEN_EXPIRED' });
  }
  return { payload: JSON.parse(record.meta) };
}

export async function consumeToken(rawToken: string, purpose: TokenPurpose, env: Env) {
  const tokenHash = await hmacSha256(env.MASTER_KEY, rawToken);
  await env.DB.prepare(
    `UPDATE access_tokens SET used_at = ?1 WHERE token_hash = ?2`
  ).bind(new Date().toISOString(), tokenHash).run();
}
```

### 3.3 一回限りトークン

`consumeToken()` を検証直後に必ず呼び、`used_at` を設定。再使用時は `TOKEN_REUSED`。

### 3.3a メンバー招待トークン(activation)+ SCR-031 有効化

招待時 (`POST /projects/{id}/members`) に `purpose='activation'` で発行し、`meta` に `invitedProjectId` / `invitedRole` を JSON 保持。SCR-031 「登録を完了する」押下時に `verifyToken` → 同一トランザクションでユーザー有効化 + `consumeToken` を実行する。

```ts
// 招待時(POST /projects/{id}/members)
const rawToken = await generateToken('activation', {
  userId,
  contractOwnerUserId,
  invitedProjectId: projectId,
  invitedRole,                  // 'admin' | 'member'
}, env, /* 7d */ 7 * 86400);
const activationUrl = `${env.WEB_BASE}/auth/activate?token=${encodeURIComponent(rawToken)}`;
await enqueueEmail({ template: 'TPL-ADMIN_USER_REGISTER', activationUrl, ... });

// SCR-031 で「登録を完了する」押下時(POST /auth/invitations/{token}/activate)
const { payload } = await verifyToken(rawToken, 'activation', env);
const { userId, invitedProjectId, invitedRole } = payload;
await env.DB.batch([
  env.DB.prepare(
    `UPDATE users SET name = ?1, password_hash = ?2, status = 'active', updated_at = ?3 WHERE id = ?4`
  ).bind(displayName, passwordHash, now, userId),
  env.DB.prepare(
    `INSERT INTO terms_agreements (id, user_id, doc_type, terms_version, agreed_at) VALUES (?1, ?2, 'terms_of_service', ?3, ?4), (?5, ?2, 'privacy_policy', ?6, ?4)`
  ).bind(ulid(), userId, termsVer, now, ulid(), privacyVer),
  env.DB.prepare(
    `UPDATE project_users SET valid = 1, updated_at = ?1 WHERE user_id = ?2 AND project_id = ?3`
  ).bind(now, userId, invitedProjectId),
]);
await consumeToken(rawToken, 'activation', env);
await writeAudit({ action: 'user.activation_completed', actor_user_id: userId, target_user_id: userId, meta: { project_id: invitedProjectId, role: invitedRole } });
```

再送時 (`POST /members/{id}/resend-invitation`) は旧トークンを `consumeToken()` で失効させ、同一 `invitedProjectId` / `invitedRole` を `meta` に含めて新規トークンを発行する。

### 3.3b プロジェクト連絡先メール確認トークン(contact_verify)+ SCR-032 着地

SCR-010-M1 プロジェクト設定モーダルで連絡先メールが新規入力 / 変更された時に `purpose='contact_verify'` で発行し、`meta` に `projectId` を JSON 保持。SCR-032 着地ページから `POST /auth/contact-verifications/{token}` 経由で同一トランザクションで `projects.contact_email_verified_at` セット + `consumeToken` を実行する。

```ts
// 連絡先メール変更時(POST/PATCH /projects/{id} で contact_email が変更された場合)
const rawToken = await generateToken('contact_verify', { projectId }, env, /* 24h */ 24 * 3600);
const verifyUrl = `${env.WEB_BASE}/auth/contact-verify?token=${encodeURIComponent(rawToken)}`;
await enqueueEmail({ template: 'TPL-PROJECT_CONTACT_VERIFY', verifyUrl, ... });

// SCR-032 着地時(POST /auth/contact-verifications/{token})
const { payload } = await verifyToken(rawToken, 'contact_verify', env);
const { projectId } = payload;
await env.DB.prepare(
  `UPDATE projects SET contact_email_verified_at = ?1, updated_at = ?1 WHERE id = ?2`
).bind(now, projectId).run();
await consumeToken(rawToken, 'contact_verify', env);
await writeAudit({ action: 'project.contact_email_verified', target_project_id: projectId, meta: { email_hmac } });
```

Turnstile 不要(状態変更は単純な `verified_at` セットのみで副作用が軽微、トークン HMAC-SHA256 検証 + IP/トークン単位のレート制限で十分)。再送は連絡先メール再設定 / 再保存時の自動発行のみで、明示的な「再送」操作 API は提供しない(MVP)。
